} from "./firebase";

} from "firebase/auth";

import {

  User as UserIcon,

// Types
interface ExecutionLogEntry {
  timestamp: string;
  message: string;
  type: "info" | "success" | "error";
}

interface TradeSignal {
  id: string;
  userId: string;
  instrument: string;
  strikePrice: string;
  side: "BUY" | "SELL";
  type: "LIMIT" | "MARKET";
  entryPrice: number;
  stopLoss: number;
  target: number;
  lots: number;
  status: "pending" | "triggered" | "hit_target" | "hit_sl";
  timestamp: string;
  executionTimestamp?: string;
  orderId?: string;
  executionLog?: ExecutionLogEntry[];
  isHistory: boolean;
}

interface UserSettings {
  tradingViewSecret: string;
  dhanClientId?: string;
  dhanAccessToken?: string;
  displayName?: string;
  phoneNumber?: string;
  tradingExperience?: string;
  onboardingComplete?: boolean;
  lastWebhookReceived?: string;
  lastWebhookError?: {
    message: string;
    timestamp: string;
  };
}

interface WebhookLog {
  id: string;
  userId?: string;
  timestamp: string;
  status: "success" | "error";
  instrument?: string;
  error?: string;
  payload: any;
  ip?: string;
}

export default function App() {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);
  const [settings, setSettings] = useState<UserSettings | null>(null);
  const [dhanClientId, setDhanClientId] = useState("");
  const [dhanAccessToken, setDhanAccessToken] = useState("");
  const [signals, setSignals] = useState<TradeSignal[]>([]);
  const [webhookLogs, setWebhookLogs] = useState<WebhookLog[]>([]);
  const [rawTraffic, setRawTraffic] = useState<any[]>([]);
  const [activeTab, setActiveTab] = useState<"active" | "history" | "settings">("active");
  const [expandedSignals, setExpandedSignals] = useState<Set<string>>(new Set());
  const [signalOrderTypes, setSignalOrderTypes] = useState<Record<string, "LIMIT" | "MARKET">>({});
  const [isSaving, setIsSaving] = useState(false);
  const [isTesting, setIsTesting] = useState(false);
  const [settingsLoading, setSettingsLoading] = useState(true);
  const [authMode, setAuthMode] = useState<"login" | "signup">("login");
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [onboardingData, setOnboardingData] = useState({
    name: "",
    phone: "",
    experience: "Beginner"
  });
  const [notification, setNotification] = useState<{
    message: string;
    type: "success" | "error";
  } | null>(null);
  const [confirmModal, setConfirmModal] = useState<{
    isOpen: boolean;
    type: "status_update" | "execute_trade";
    signalId: string;
    status?: TradeSignal["status"];
    instrument: string;
    signal?: TradeSignal;
    orderType?: "LIMIT" | "MARKET";
  }>({
    isOpen: false,
    type: "status_update",
    signalId: "",
    instrument: "",
  });
  const [error, setError] = useState<string | null>(null);
  const [trafficError, setTrafficError] = useState<string | null>(null);
  const [networkHeartbeat, setNetworkHeartbeat] = useState<any>(null);
  const [serverStatus, setServerStatus] = useState<{ status: string; message: string; config?: any } | null>(null);
  const [isConnected, setIsConnected] = useState(true);

  // Filtering and Sorting State
  const [searchTerm, setSearchTerm] = useState("");
  const [statusFilter, setStatusFilter] = useState<string>("all");
  const [sortBy, setSortBy] = useState<"timestamp" | "entryPrice" | "status">("timestamp");
  const [sortOrder, setSortOrder] = useState<"asc" | "desc">("desc");
  const [dateRange, setDateRange] = useState({ start: "", end: "" });
  const [instrumentDetails, setInstrumentDetails] = useState<Record<string, InstrumentDetails>>({});
  const [isFetchingDetails, setIsFetchingDetails] = useState<Record<string, boolean>>({});
  const [livePrices, setLivePrices] = useState<Record<string, PriceUpdate>>({});

  // Live Price Subscriptions
  useEffect(() => {
    if (!user || signals.length === 0) return;

    // Get unique instruments from active signals
    const activeInstruments = Array.from(new Set(
      signals
        .filter(s => !s.isHistory)
        .map(s => s.instrument)
    )) as string[];

    const unsubscribers: (() => void)[] = [];

    activeInstruments.forEach((symbol: string) => {
      const callback = (update: PriceUpdate) => {
        setLivePrices(prev => ({
          ...prev,
          [symbol]: update
        }));
      };
      
      marketDataService.subscribe(symbol, callback);
      unsubscribers.push(() => marketDataService.unsubscribe(symbol, callback));
    });

    return () => {
      unsubscribers.forEach(unsub => unsub());
    };
  }, [user, signals]);

  const showNotification = (title: string, body: string) => {
    // Browser Notification
    if ("Notification" in window && Notification.permission === "granted") {
      try {
        new Notification(title, { body });
      } catch (e) {
        console.warn("Browser notification failed:", e);
      }
    }
    // App Toast
    setNotification({ message: `${title}: ${body}`, type: "success" });
    setTimeout(() => setNotification(null), 5000);
  };

  // Auth listener
  useEffect(() => {
    // Request notification permission
    if ("Notification" in window && Notification.permission === "default") {
      Notification.requestPermission();
    }

    const unsubscribe = onAuthStateChanged(auth, (u) => {
      setUser(u);
      if (u) {
        setOnboardingData(prev => ({
          ...prev,
          name: u.displayName || "",
        }));
      }
      setLoading(false);
    });

    const handleOnline = () => setIsConnected(true);
    const handleOffline = () => setIsConnected(false);

    window.addEventListener("online", handleOnline);
    window.addEventListener("offline", handleOffline);

    return () => {
      unsubscribe();
      window.removeEventListener("online", handleOnline);
      window.removeEventListener("offline", handleOffline);
    };
  }, []);

  // Real-time market data subscription
  useEffect(() => {
    if (!signals.length) return;

    const activeInstruments = new Set(
      signals
        .filter(s => !s.isHistory && (s.status === 'pending' || s.status === 'triggered'))
        .map(s => s.instrument.toUpperCase())
    );

    const handlePriceUpdate = (update: PriceUpdate) => {
      setLivePrices(prev => ({
        ...prev,
        [update.symbol]: update
      }));
    };

    activeInstruments.forEach((symbol: string) => {
      marketDataService.subscribe(symbol, handlePriceUpdate);
    });

    return () => {
      activeInstruments.forEach((symbol: string) => {
        marketDataService.unsubscribe(symbol, handlePriceUpdate);
      });
    };
  }, [signals]);

  // Fetch settings and signals
  useEffect(() => {
    if (!user) return;

    let isFirstLoad = true;

    const settingsRef = doc(db, "users", user.uid);
    const unsubSettings = onSnapshot(settingsRef, (docSnap) => {
      if (docSnap.exists()) {
        const data = docSnap.data() as UserSettings;
        setSettings(data);
        setDhanClientId(data.dhanClientId || "");
        setDhanAccessToken(data.dhanAccessToken || "");
        
        // Ensure webhook_secrets is populated for existing users
        if (data.tradingViewSecret) {
          const secretRef = doc(db, "webhook_secrets", data.tradingViewSecret);
          getDoc(secretRef).then((secretSnap) => {
            if (!secretSnap.exists() || secretSnap.data()?.userId !== user.uid) {
              setDoc(secretRef, { userId: user.uid })
                .catch(e => handleFirestoreError(e, OperationType.WRITE, "webhook_secrets"));
            }
          });
        }
      } else {
        // Initialize settings if not exists
        const newSecret = Math.random().toString(36).substring(2, 15) + Math.random().toString(36).substring(2, 15);
        setDoc(settingsRef, { tradingViewSecret: newSecret })
          .then(() => {
            // Also populate webhook_secrets for unauthenticated server access
            return setDoc(doc(db, "webhook_secrets", newSecret), { userId: user.uid });
          })
          .catch(e => handleFirestoreError(e, OperationType.WRITE, "users/webhook_secrets"));
      }
      setSettingsLoading(false);
    }, (e) => {
      setSettingsLoading(false);
      handleFirestoreError(e, OperationType.GET, "users");
    });

    const signalsQuery = query(
      collection(db, "signals"),
      where("userId", "==", user.uid),
      orderBy("timestamp", "desc")
    );

    const unsubSignals = onSnapshot(signalsQuery, (snapshot) => {
      if (isFirstLoad) {
        isFirstLoad = false;
      } else {
        snapshot.docChanges().forEach((change) => {
          const signal = { id: change.doc.id, ...change.doc.data() } as TradeSignal;
          if (change.type === "added" && !signal.isHistory) {
            showNotification("New Signal Received", `${signal.instrument} @ ${signal.entryPrice}`);
          } else if (change.type === "modified") {
            const statusLabel = signal.status.replace('_', ' ').toUpperCase();
            showNotification("Signal Updated", `${signal.instrument} is now ${statusLabel}`);
          }
        });
      }

      const fetchedSignals = snapshot.docs.map(d => ({ id: d.id, ...d.data() } as TradeSignal));
      
      // Auto-move to history logic (every other day)
      // For simplicity, we move signals older than 24 hours to history
      const now = new Date();
      fetchedSignals.forEach(async (signal) => {
        const signalDate = new Date(signal.timestamp);
        const diffHours = (now.getTime() - signalDate.getTime()) / (1000 * 60 * 60);
        if (diffHours > 24 && !signal.isHistory) {
          await updateDoc(doc(db, "signals", signal.id), { isHistory: true });
        }
      });

      setSignals(fetchedSignals);
    }, (e) => handleFirestoreError(e, OperationType.GET, "signals"));

    // Fetch Webhook Logs
    const logsQuery = query(
      collection(db, "webhook_logs"),
      where("userId", "==", user.uid)
    );

    const unsubLogs = onSnapshot(logsQuery, (snapshot) => {
      const fetchedLogs = snapshot.docs
        .map(d => ({ id: d.id, ...d.data() } as WebhookLog))
        .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());
      setWebhookLogs(fetchedLogs);
    }, (e) => {
      console.warn("Logs fetch failed:", e);
    });

    // Fetch Raw Traffic (for admin)
    const trafficQuery = query(
      collection(db, "raw_webhook_traffic"),
      orderBy("timestamp", "desc")
    );

    const unsubTraffic = onSnapshot(trafficQuery, (snapshot) => {
      const fetchedTraffic = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
      setRawTraffic(fetchedTraffic);
      setTrafficError(null);
    }, (e) => {
      console.warn("Traffic fetch failed (likely not admin):", e);
      setTrafficError(e.message);
    });

    const unsubStatus = onSnapshot(doc(db, "server_status", "network_heartbeat"), (d) => {
      if (d.exists()) setNetworkHeartbeat(d.data());
    });

    return () => {
      unsubSettings();
      unsubSignals();
      unsubLogs();
      unsubTraffic();
      unsubStatus();
    };
  }, [user]);

  const filteredSignals = useMemo(() => {
    return signals
      .filter(s => {
        const matchesSearch = s.instrument.toLowerCase().includes(searchTerm.toLowerCase());
        const matchesStatus = statusFilter === "all" || s.status === statusFilter;
        const matchesTab = activeTab === "active" ? !s.isHistory : s.isHistory;
        
        let matchesDate = true;
        if (dateRange.start) {
          matchesDate = matchesDate && new Date(s.timestamp) >= new Date(dateRange.start);
        }
        if (dateRange.end) {
          const endDate = new Date(dateRange.end);
          endDate.setDate(endDate.getDate() + 1);
          matchesDate = matchesDate && new Date(s.timestamp) <= endDate;
        }

        return matchesSearch && matchesStatus && matchesTab && matchesDate;
      })
      .sort((a, b) => {
        let comparison = 0;
        if (sortBy === "timestamp") {
          comparison = new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime();
        } else if (sortBy === "entryPrice") {
          comparison = a.entryPrice - b.entryPrice;
        } else if (sortBy === "status") {
          comparison = a.status.localeCompare(b.status);
        }
        return sortOrder === "asc" ? comparison : -comparison;
      });
  }, [signals, searchTerm, statusFilter, activeTab, sortBy, sortOrder, dateRange]);

  const activeSignalsCount = useMemo(() => signals.filter(s => !s.isHistory).length, [signals]);
  const historySignalsCount = useMemo(() => signals.filter(s => s.isHistory).length, [signals]);

  const activeSignals = useMemo(() => signals.filter(s => !s.isHistory), [signals]);
  const historySignals = useMemo(() => signals.filter(s => s.isHistory), [signals]);

  const stats = useMemo(() => {
    const completed = historySignals.filter(s => s.status === "hit_target" || s.status === "hit_sl");
    const wins = completed.filter(s => s.status === "hit_target").length;
    const total = completed.length;
    const winRate = total > 0 ? (wins / total) * 100 : 0;
    return { wins, total, winRate };
  }, [historySignals]);

  useEffect(() => {
    if (notification) {
      const timer = setTimeout(() => setNotification(null), 5000);
      return () => clearTimeout(timer);
    }
  }, [notification]);

  const handleStatusUpdate = async (id: string, status: TradeSignal["status"], logMessage?: string) => {
    try {
      const signalRef = doc(db, "signals", id);
      const signalSnap = await getDoc(signalRef);
      const currentSignal = signalSnap.data() as TradeSignal;
      
      const updateData: any = { status };
      const now = new Date().toISOString();
      
      if (status === "triggered") {
        updateData.executionTimestamp = now;
      }

      // Add to execution log
      const newLogEntry: ExecutionLogEntry = {
        timestamp: now,
        message: logMessage || `Status changed to ${status.replace('_', ' ')}`,
        type: status === "hit_target" ? "success" : status === "hit_sl" ? "error" : "info"
      };

      const currentLog = currentSignal.executionLog || [];
      updateData.executionLog = [...currentLog, newLogEntry];
      
      await updateDoc(signalRef, updateData);
      setConfirmModal(prev => ({ ...prev, isOpen: false }));
      
      const statusLabel = status.replace('_', ' ').toUpperCase();
      setNotification({ message: `Signal updated to ${statusLabel}`, type: "success" });
    } catch (e) {
      setNotification({ message: "Failed to update signal status. Please check your connection.", type: "error" });
      handleFirestoreError(e, OperationType.UPDATE, "signals");
    }
  };

  const saveUserSettings = async () => {
    if (!user) return;
    setIsSaving(true);
    try {
      await updateDoc(doc(db, "users", user.uid), { 
        dhanClientId,
        dhanAccessToken
      });
      setNotification({ message: "Settings saved successfully!", type: "success" });
    } catch (e) {
      setNotification({ message: "Failed to save settings. Please try again.", type: "error" });
      handleFirestoreError(e, OperationType.UPDATE, "users");
    } finally {
      setIsSaving(false);
    }
  };

  const openDhan = (signal: TradeSignal, orderType: "LIMIT" | "MARKET" = "LIMIT") => {
    try {
      // Validation before opening
      if (!signal.entryPrice || !signal.stopLoss || !signal.target) {
        setNotification({ message: "Incomplete signal data. Check your TradingView alert template.", type: "error" });
        return;
      }

      if (!settings?.dhanClientId) {
        setNotification({ message: "Please set your Dhan Client ID in Settings first.", type: "error" });
        setActiveTab("settings");
        return;
      }

      const symbol = signal.instrument + (signal.strikePrice ? ` ${signal.strikePrice}` : "");
      const quantity = signal.lots * 50; 
      const side = "BUY";
      const type = orderType;
      const price = signal.entryPrice;
      
      const deepLink = `dhan://order?symbol=${encodeURIComponent(symbol)}&price=${price}&quantity=${quantity}&side=${side}&type=${type}&sl=${signal.stopLoss}&target=${signal.target}&clientId=${settings.dhanClientId}`;
      
      // Attempt to open deep link
      const start = Date.now();
      const timeout = 2000;
      
      // Create a hidden iframe to try and open the deep link
      // This is a common pattern to avoid the "Address invalid" error in some browsers
      const iframe = document.createElement("iframe");
      iframe.style.display = "none";
      iframe.src = deepLink;
      document.body.appendChild(iframe);
      
      setTimeout(() => {
        document.body.removeChild(iframe);
        const elapsed = Date.now() - start;
        
        if (elapsed < timeout + 100) {
          // If we're still here quickly, it might have failed
          // Note: This is not 100% reliable due to browser security
          setNotification({ 
            message: "Dhan app could not be opened. Please check if the app is installed and you are on a mobile device.", 
            type: "error" 
          });
        } else {
          setNotification({ message: "Dhan app requested. Please check your phone.", type: "success" });
          if (signal.status === "pending") {
            handleStatusUpdate(signal.id, "triggered");
          }
        }
      }, timeout);

      // Fallback: direct location change if iframe fails or for better compatibility
      window.location.href = deepLink;

    } catch (err) {
      console.error("Dhan deep link error:", err);
      setNotification({ message: "An error occurred while trying to open the Dhan app.", type: "error" });
    }
  };

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
    setNotification({ message: "Copied to clipboard!", type: "success" });
  };

  const clearRawTraffic = async () => {
    try {
      const promises = rawTraffic.map(log => deleteDoc(doc(db, "raw_webhook_traffic", log.id)));
      await Promise.all(promises);
      setNotification({ message: "Raw traffic logs cleared", type: "success" });
    } catch (error) {
      handleFirestoreError(error, OperationType.DELETE, "raw_webhook_traffic");
    }
  };

  const toggleExpand = async (id: string, instrument: string) => {
    setExpandedSignals(prev => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else {
        next.add(id);
        // Fetch details if not already fetched
        if (!instrumentDetails[instrument] && !isFetchingDetails[instrument]) {
          setIsFetchingDetails(curr => ({ ...curr, [instrument]: true }));
          getInstrumentDetails(instrument).then(details => {
            setInstrumentDetails(curr => ({ ...curr, [instrument]: details }));
            setIsFetchingDetails(curr => ({ ...curr, [instrument]: false }));
          });
        }
      }
      return next;
    });
  };

  const [isCheckingServer, setIsCheckingServer] = useState(false);

  const checkServerConnection = async () => {
    setIsCheckingServer(true);
    try {
      const response = await fetch("/api/test-db");
      const data = await response.json();
      setServerStatus(data);
      if (data.status === "success") {
        setNotification({ message: "Server connection to Firestore is healthy!", type: "success" });
      } else {
        setNotification({ message: `Server Error: ${data.message}`, type: "error" });
      }
    } catch (error) {
      setServerStatus({ status: "error", message: "Could not reach the server." });
      setNotification({ message: "Server is unreachable.", type: "error" });
    } finally {
      setIsCheckingServer(false);
    }
  };

  const forceAdmin = async () => {
    if (!user) return;
    try {
      await updateDoc(doc(db, "users", user.uid), { role: "admin" });
      setNotification({ message: "Admin role granted! Refreshing...", type: "success" });
      setTimeout(() => window.location.reload(), 1500);
    } catch (e) {
      handleFirestoreError(e, OperationType.WRITE, "users");
    }
  };

  const deleteTestSignals = async () => {
    if (!user) return;
    const testSignals = signals.filter(s => s.instrument.startsWith("TEST_"));
    if (testSignals.length === 0) {
      setNotification({ message: "No test signals found to delete.", type: "success" });
      return;
    }

    try {
      const deletePromises = testSignals.map(s => deleteDoc(doc(db, "signals", s.id)));
      await Promise.all(deletePromises);
      setNotification({ message: `Deleted ${testSignals.length} test signals.`, type: "success" });
    } catch (e) {
      setNotification({ message: "Failed to delete test signals.", type: "error" });
      handleFirestoreError(e, OperationType.DELETE, "signals");
    }
  };

  const [isTestingWebhook, setIsTestingWebhook] = useState(false);

  const testWebhook = async () => {
    if (!settings?.tradingViewSecret) {
      setNotification({ message: "Please generate a secret first.", type: "error" });
      return;
    }
    setIsTestingWebhook(true);
    try {
      const response = await fetch(`/api/webhook/tradingview/${settings.tradingViewSecret}`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          instrument: "TEST_SIGNAL",
          entryPrice: 100,
          stopLoss: 90,
          target: 110,
          lots: 1,
          side: "BUY",
          type: "MARKET"
        })
      });
      const data = await response.json();
      if (data.status === "success") {
        setNotification({ message: "Test webhook sent successfully! Check your signals.", type: "success" });
      } else {
        setNotification({ message: `Test failed: ${data.message}`, type: "error" });
      }
    } catch (error) {
      setNotification({ message: "Failed to send test webhook.", type: "error" });
    } finally {
      setIsTestingWebhook(false);
    }
  };

  const handleEmailAuth = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsSaving(true);
    try {
      if (authMode === "signup") {
        await createUserWithEmailAndPassword(auth, email, password);
        setNotification({ message: "Account created! Please complete your profile.", type: "success" });
      } else {
        await signInWithEmailAndPassword(auth, email, password);
        setNotification({ message: "Welcome back!", type: "success" });
      }
    } catch (e: any) {
      setNotification({ message: e.message || "Authentication failed", type: "error" });
    } finally {
      setIsSaving(false);
    }
  };

  const handleOnboarding = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!user) return;
    setIsSaving(true);
    try {
      await updateProfile(user, { displayName: onboardingData.name });
      await updateDoc(doc(db, "users", user.uid), {
        displayName: onboardingData.name,
        phoneNumber: onboardingData.phone,
        tradingExperience: onboardingData.experience,
        onboardingComplete: true
      });
      setNotification({ message: "Profile completed!", type: "success" });
    } catch (e: any) {
      setNotification({ message: "Failed to save profile", type: "error" });
    } finally {
      setIsSaving(false);
    }
  };

  if (loading || (user && settingsLoading)) {
    return (
      <div className="min-h-screen bg-slate-950 flex items-center justify-center">
        <RefreshCw className="w-8 h-8 text-blue-500 animate-spin" />
      </div>
    );
  }

  if (!user) {
    return (
      <div className="min-h-screen bg-slate-950 flex flex-col items-center justify-center p-4">
        <motion.div 
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="max-w-md w-full bg-slate-900 border border-slate-800 rounded-2xl p-8 shadow-2xl"
        >
          <div className="text-center mb-8">
            <TrendingUp className="w-16 h-16 text-blue-500 mx-auto mb-6" />
            <h1 className="text-3xl font-bold text-white mb-2">DhanBridge</h1>
            <p className="text-slate-400">Automate your trading signals from TradingView directly to your Dhan account.</p>
          </div>

                          className="w-full py-1.5 bg-slate-900 hover:bg-slate-800 disabled:opacity-50 text-[9px] font-bold text-slate-400 uppercase rounded-lg border border-slate-800 transition-all flex items-center justify-center gap-2"
                        >
                          {isCheckingServer ? <RefreshCw className="w-3 h-3 animate-spin" /> : <Globe className="w-3 h-3" />}
                          Check Server Connection
                        </button>
                        {settings?.role !== 'admin' && (
                          <button 
                            onClick={forceAdmin}
                            className="w-full mt-2 py-1 bg-amber-950/20 hover:bg-amber-950/40 text-[8px] font-bold text-amber-600 uppercase rounded-lg border border-amber-900/30 transition-all"
                          >
                            Enable Admin Mode (Debug)
                          </button>
                        )}
                        {serverStatus && (
                          <p className={`mt-1 text-[8px] text-center font-mono ${serverStatus.status === 'success' ? 'text-green-500' : 'text-red-500'}`}>
                            {serverStatus.message}
                          </p>
                        )}
                      </div>
                      <div className="flex justify-between text-[10px] font-mono pt-2 border-t border-slate-800/50">
                        <span className="text-slate-500">Last Webhook:</span>
                        <span className={settings?.lastWebhookReceived ? "text-green-400" : "text-slate-500"}>
                          {settings?.lastWebhookReceived ? new Date(settings.lastWebhookReceived).toLocaleString() : "Never Received"}
                        </span>
                      </div>
                      {!settings?.lastWebhookReceived && !settings?.lastWebhookError && (
                        <div className="mt-2 p-2 bg-amber-500/10 border border-amber-500/20 rounded-lg">
                          <p className="text-[10px] text-amber-200 leading-relaxed">
                            <span className="font-bold">No signals detected yet.</span> If you've sent an alert:
                            <br/>1. Check if your TradingView plan supports Webhooks.
                            <br/>2. Verify the Webhook URL and Secret match exactly.
                            <br/>3. Check the "Alerts Log" in TradingView for errors.
                          </p>
                        </div>
                      )}
                      {settings?.lastWebhookError && (
                        <div className="flex flex-col gap-1 pt-2 border-t border-slate-800/50">
                          <div className="flex items-center justify-between">
                            <span className="text-[10px] font-mono text-red-400 font-bold uppercase">Last Error:</span>
                            <button 
                              onClick={async () => {
                                if (!user) return;
                                await updateDoc(doc(db, "users", user.uid), { lastWebhookError: null });
                              }}
                              className="text-[8px] font-bold text-slate-500 hover:text-slate-300 uppercase tracking-tighter"
                            >
                              Clear
                            </button>
                          </div>
                          <span className="text-[10px] font-mono text-red-300 leading-relaxed">
                            {settings.lastWebhookError.message}
                          </span>
                          <span className="text-[8px] font-mono text-slate-600">
                            {new Date(settings.lastWebhookError.timestamp).toLocaleString()}
                          </span>
                        </div>
                      )}
                    </div>

                    <div className="pt-4">
                      <div className="flex items-center justify-between mb-3">
                        <h4 className="text-[10px] font-bold text-slate-500 uppercase tracking-wider flex items-center gap-2">
                          <List className="w-3 h-3" />
                          Recent Webhook Activity
                        </h4>
                        {webhookLogs.length > 0 && (
                          <button 
                            onClick={async () => {
                              if (!user) return;
                              const deletePromises = webhookLogs.map(l => deleteDoc(doc(db, "webhook_logs", l.id)));
                              await Promise.all(deletePromises);
                            }}
                            className="text-[8px] font-bold text-red-500 hover:text-red-400 uppercase tracking-tighter"
                          >
                            Clear Logs
                          </button>
                        )}
                      </div>
                      <div className="bg-slate-950/50 border border-slate-800/50 rounded-xl overflow-hidden">
                        <div className="max-h-[200px] overflow-y-auto custom-scrollbar">
                          {webhookLogs.length === 0 ? (
                            <div className="p-6 text-center">
                              <p className="text-[10px] text-slate-600 italic">No activity logged yet.</p>
                            </div>
                          ) : (
                            <div className="divide-y divide-slate-800/50">
                              {webhookLogs.map((log) => (
                                <div key={log.id} className="p-2.5 hover:bg-slate-900/50 transition-colors">
                                  <div className="flex items-center justify-between mb-1">
                                    <div className="flex items-center gap-1.5">
                                      <div className={`w-1 h-1 rounded-full ${log.status === 'success' ? 'bg-green-500' : 'bg-red-500'}`} />
                                      <span className={`text-[9px] font-bold uppercase ${log.status === 'success' ? 'text-green-500' : 'text-red-500'}`}>
                                        {log.status === 'success' ? 'Delivered' : 'Rejected'}
                                      </span>
                                    </div>
                                    <span className="text-[8px] font-mono text-slate-600">
                                      {new Date(log.timestamp).toLocaleTimeString()}
                                    </span>
                                  </div>
                                  <div className="flex flex-col gap-0.5">
                                    <p className="text-[9px] font-mono text-slate-400 truncate">
                                      {log.status === 'success' ? `Instrument: ${log.instrument}` : `Error: ${log.error}`}
                                    </p>
                                    <details className="mt-0.5">
                                      <summary className="text-[8px] font-bold text-slate-600 cursor-pointer hover:text-slate-400 uppercase tracking-tighter list-none flex items-center gap-1">
                                        <span>View Payload</span>
                                        <ChevronDown className="w-2 h-2" />
                                      </summary>
                                      <pre className="mt-1.5 p-2 bg-slate-950 rounded border border-slate-800/50 text-[8px] font-mono text-blue-400/60 overflow-x-auto">
                                        {JSON.stringify(log.payload, null, 2)}
                                      </pre>
                                    </details>
                                  </div>
                                </div>
                              ))}
                            </div>
                          )}
                        </div>
                      </div>
                    </div>

                    <div className="pt-4 border-t border-slate-800/50 mt-4">
                      <div className="p-3 bg-amber-500/5 border border-amber-500/20 rounded-xl mb-4">
                        <h4 className="text-[10px] font-bold text-amber-400 uppercase mb-2 flex items-center gap-1">
                          <AlertCircle className="w-3 h-3" />
                          Delivery Debugger
                        </h4>
                        <p className="text-[9px] text-slate-400 leading-relaxed mb-3">
                          If TradingView says "Delivered" but you see nothing below, use this Public Echo URL in your TradingView alert to bypass all security and verify the connection:
                        </p>
                        <div className="flex gap-2">
                          <input 
                            type="text" 
                            readOnly 
                            value={`${window.location.origin}/api/debug/echo`}
                            className="flex-1 bg-black/50 border border-slate-800 rounded-lg px-3 py-1.5 text-[9px] font-mono text-amber-200 focus:outline-none"
                          />
                          <button 
                            onClick={() => copyToClipboard(`${window.location.origin}/api/debug/echo`)}
                            className="p-1.5 bg-slate-800 hover:bg-slate-700 rounded-lg text-slate-400 transition-colors"
                          >
                            <Copy className="w-3.5 h-3.5" />
                          </button>
                        </div>
                        <p className="mt-2 text-[8px] text-slate-500 italic">
                          * Note: This URL will ONLY log the data for debugging. It will NOT create signals.
                        </p>
                      </div>

                      <div className="flex items-center justify-between mb-3">
                        <h4 className="text-[10px] font-bold text-amber-500 uppercase tracking-wider flex items-center gap-2">
                          <Globe className="w-3 h-3" />
                          Raw Webhook Traffic (Debug)
                        </h4>
                        {trafficError && (
                          <span className="text-[8px] text-red-500 font-mono">
                            Error: {trafficError} (Check if you are Admin)
                          </span>
                        )}
                        {rawTraffic.length > 0 && (
                          <button 
                            onClick={clearRawTraffic}
                            className="text-[8px] font-bold text-red-500 hover:text-red-400 uppercase tracking-tighter"
                          >
                            Clear Debug Logs
                          </button>
                        )}
                      </div>
                      <div className="bg-slate-950/50 border border-slate-800/50 rounded-xl overflow-hidden">
                        <div className="max-h-[200px] overflow-y-auto custom-scrollbar">
                          {rawTraffic.length === 0 ? (
                            <div className="p-6 text-center">
                              <p className="text-[10px] text-slate-600 italic">No raw traffic detected yet.</p>
                            </div>
                          ) : (
                            <div className="divide-y divide-slate-800/50">
                              {rawTraffic.map((log) => (
                                <div key={log.id} className="p-2.5 hover:bg-slate-900/50 transition-colors">
                                  <div className="flex items-center justify-between mb-1">
                                    <div className="flex items-center gap-1.5">
                                      <div className={`w-1 h-1 rounded-full ${
                                        log.status === 'processed' ? 'bg-green-500' : 
                                        log.status === 'rejected' ? 'bg-amber-500' : 'bg-blue-500'
                                      }`} />
                                      <span className={`text-[9px] font-bold uppercase ${
                                        log.status === 'processed' ? 'text-green-500' : 
                                        log.status === 'rejected' ? 'text-amber-500' : 'text-blue-500'
                                      }`}>
                                        {log.status}
                                      </span>
                                    </div>
                                    <span className="text-[8px] font-mono text-slate-600">
                                      {new Date(log.timestamp).toLocaleTimeString()}
                                    </span>
                                  </div>
                                  <div className="flex flex-col gap-0.5">
                                    <p className="text-[9px] font-mono text-slate-400 truncate">
                                      IP: {log.ip} {log.reason ? `| Reason: ${log.reason}` : ''}
                                    </p>
                                    {log.receivedSecret && (
                                      <p className="text-[9px] font-mono text-amber-500/80 mt-0.5">
                                        Received Secret: <span className="bg-amber-500/10 px-1 rounded">{log.receivedSecret}</span>
                                      </p>
                                    )}
                                    <details className="mt-0.5">
                                      <summary className="text-[8px] font-bold text-slate-600 cursor-pointer hover:text-slate-400 uppercase tracking-tighter list-none flex items-center gap-1">
                                        <span>View Raw Data</span>
                                        <ChevronDown className="w-2 h-2" />
                                      </summary>
                                      <div className="mt-1.5 space-y-2">
                                        <div>
                                          <p className="text-[7px] text-slate-500 uppercase font-bold">Headers</p>
                                          <pre className="p-1.5 bg-slate-950 rounded border border-slate-800/50 text-[8px] font-mono text-slate-500 overflow-x-auto">
                                            {JSON.stringify(log.headers, null, 2)}
                                          </pre>
                                        </div>
                                        <div>
                                          <p className="text-[7px] text-slate-500 uppercase font-bold">Body</p>
                                          <pre className="p-1.5 bg-slate-950 rounded border border-slate-800/50 text-[8px] font-mono text-blue-400/60 overflow-x-auto">
                                            {log.rawBody}
                                          </pre>
                                        </div>
                                      </div>
                                    </details>
                                  </div>
                                </div>
                              ))}
                            </div>
                          )}
                        </div>
                      </div>
                    </div>
                  </div>

                  <div className="pt-6 border-t border-slate-800">
                    <h4 className="text-sm font-bold text-white mb-4 flex items-center gap-2">
                      <RefreshCw className={`w-4 h-4 ${isTesting ? 'animate-spin' : ''}`} />
                      Test Connection
                    </h4>
                    <button 
                      onClick={testWebhook}
                      disabled={isTesting}
                      className="w-full bg-slate-800 hover:bg-slate-700 text-white font-bold py-3 rounded-xl transition-all border border-slate-700 flex items-center justify-center gap-2"
                    >
                      {isTesting ? "Sending Test Signal..." : "Send Test Signal"}
                    </button>
                    <p className="mt-2 text-xs text-slate-500 text-center">This will send a sample signal to your dashboard to verify real-time updates.</p>
                  </div>

                  <div className="pt-6 border-t border-slate-800">
                    <h4 className="text-sm font-bold text-red-500 mb-4 flex items-center gap-2">
                      <Trash2 className="w-4 h-4" />
                      Data Management
                    </h4>
                    <button 
                      onClick={deleteTestSignals}
                      className="w-full bg-red-500/10 hover:bg-red-500/20 text-red-500 font-bold py-3 rounded-xl transition-all border border-red-500/20 flex items-center justify-center gap-2"
                    >
                      Delete All Test Signals
                    </button>
                    <p className="mt-2 text-xs text-slate-500 text-center">Removes all signals starting with "TEST_" from your history and active list.</p>
                  </div>
                </div>

                <div className="bg-blue-500/5 border border-blue-500/10 p-4 rounded-xl">
                  <h4 className="text-sm font-bold text-blue-400 mb-2 flex items-center gap-2">
                    <AlertCircle className="w-4 h-4" />
                    How to use with TradingView
                  </h4>
                  <ol className="text-xs text-slate-400 space-y-2 list-decimal ml-4">
                    <li>Create an Alert on TradingView.</li>
                    <li>Set the <strong>Webhook URL</strong> to the one above.</li>
                    <li>In the <strong>Message</strong> field, paste the following JSON:</li>
                  </ol>
                  <div className="relative mt-3">
                    <pre className="bg-slate-950 p-3 rounded-lg text-[10px] font-mono text-blue-300 overflow-x-auto">
{`{
  "secret": "${settings?.tradingViewSecret}",
  "instrument": "{{ticker}}",
  "strikePrice": "19500 CE",
  "side": "BUY",
  "type": "LIMIT",
  "entryPrice": {{close}},
  "stopLoss": 19400,
  "target": 19700,
  "lots": 1
}`}
                    </pre>
                    <button 
                      onClick={() => copyToClipboard(`{
  "secret": "${settings?.tradingViewSecret}",
  "instrument": "{{ticker}}",
  "strikePrice": "19500 CE",
  "side": "BUY",
  "type": "LIMIT",
  "entryPrice": {{close}},
  "stopLoss": 19400,
  "target": 19700,
  "lots": 1
}`)}
                      className="absolute top-2 right-2 p-1.5 bg-slate-800/50 hover:bg-slate-700 rounded-md text-slate-400 transition-colors"
                      title="Copy JSON"
                    >
                      <Copy className="w-3.5 h-3.5" />
                    </button>
                  </div>
                </div>
              </div>
            </motion.div>
          )}
        </AnimatePresence>
      </main>

      {/* Confirmation Modal */}
      <AnimatePresence>
        {confirmModal.isOpen && (
          <div className="fixed inset-0 z-50 flex items-center justify-center p-4">
            <motion.div 
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              onClick={() => setConfirmModal(prev => ({ ...prev, isOpen: false }))}
              className="absolute inset-0 bg-slate-950/80 backdrop-blur-sm"
            />
            <motion.div 
              initial={{ opacity: 0, scale: 0.95, y: 20 }}
              animate={{ opacity: 1, scale: 1, y: 0 }}
              exit={{ opacity: 0, scale: 0.95, y: 20 }}
              className="relative w-full max-w-sm bg-slate-900 border border-slate-800 rounded-2xl p-6 shadow-2xl"
            >
              <div className="text-center">
                <div className={`w-16 h-16 rounded-full mx-auto mb-4 flex items-center justify-center ${
                  confirmModal.type === 'execute_trade' ? 'bg-blue-500/10 text-blue-500' :
                  confirmModal.status === 'hit_target' ? 'bg-green-500/10 text-green-500' : 'bg-red-500/10 text-red-500'
                }`}>
                  {confirmModal.type === 'execute_trade' ? <TrendingUp className="w-8 h-8" /> :
                   confirmModal.status === 'hit_target' ? <CheckCircle2 className="w-8 h-8" /> : <XCircle className="w-8 h-8" />}
                </div>
                <h3 className="text-xl font-bold text-white mb-2">
                  {confirmModal.type === 'execute_trade' ? 'Execute Trade' : 'Confirm Update'}
                </h3>
                <p className="text-slate-400 mb-6">
                  {confirmModal.type === 'execute_trade' ? (
                    <>Are you sure you want to open Dhan and execute trade for <span className="text-white font-bold">{confirmModal.instrument}</span>?</>
                  ) : (
                    <>Are you sure you want to mark <span className="text-white font-bold">{confirmModal.instrument}</span> as <span className={`font-bold ${confirmModal.status === 'hit_target' ? 'text-green-500' : 'text-red-500'}`}>
                      {confirmModal.status === 'hit_target' ? 'Hit Target' : 'Hit SL'}
                    </span>?</>
                  )}
                </p>
                <div className="flex gap-3">
                  <button 
                    onClick={() => setConfirmModal(prev => ({ ...prev, isOpen: false }))}
                    className="flex-1 bg-slate-800 hover:bg-slate-700 text-slate-300 font-bold py-3 rounded-xl transition-all"
                  >
                    Cancel
                  </button>
                  <button 
                    onClick={() => {
                      if (confirmModal.type === 'execute_trade' && confirmModal.signal) {
                        openDhan(confirmModal.signal, confirmModal.orderType);
                        handleStatusUpdate(confirmModal.signalId, 'triggered');
                      } else if (confirmModal.status) {
                        handleStatusUpdate(confirmModal.signalId, confirmModal.status);
                      }
                    }}
                    className={`flex-1 font-bold py-3 rounded-xl transition-all text-white shadow-lg ${
                      confirmModal.type === 'execute_trade' ? 'bg-blue-600 hover:bg-blue-700 shadow-blue-900/20' :
                      confirmModal.status === 'hit_target' ? 'bg-green-600 hover:bg-green-700 shadow-green-900/20' : 'bg-red-600 hover:bg-red-700 shadow-red-900/20'
                    }`}
                  >
                    Confirm
                  </button>
                </div>
              </div>
            </motion.div>
          </div>
        )}
      </AnimatePresence>
    </div>
  );
}
